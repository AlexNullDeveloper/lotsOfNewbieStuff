//класс треугольник, наследует от Shape
class Triangle:public Shape { //наследуем от Shape. из-за public наследуем все, как есть
private:
	double base, height;
public:
	Triangle(double, double); //конструктор класса
};

Triangle::Triangle(double b, double h) :base(b), height(h) { //описываем конструктор.
//base = b; height = h;
	area = 0.5*base*height; //b и h будут из main
}

//класс квадрат, наследует от Shape
class Rectangle :public Shape { //наследуем от Shape. из-за public наследуем все, как есть
private:
	double base, height;
public:
	Rectangle(double, double); //конструктор класса
};

Rectangle::Rectangle(double b, double h) :base(b), height(h) { //описываем конструктор
	area = base*height; //b и h будут из main
}

//класс круг, наследует от Shape
class Circle :public Shape { //наследуем от Shape. из-за public наследуем все, как есть
private:
	double base, height;
public:
	Circle(double); //конструктор класса
};

Circle::Circle(double b) :base(b) { //описываем конструктор
	area = base * PI; //b и h будут из main
}
